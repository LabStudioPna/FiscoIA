function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents);
    if (data.evento === 'config_email') {
      const sheetConfig = obtenerOCrearHojaConfig();
      sheetConfig.getRange('B1').setValue(data.email || '');
      asegurarTriggers(); // instala los disparadores semanal y mensual automáticamente, sin que el usuario tenga que hacerlo a mano
      return ContentService.createTextOutput(JSON.stringify({status:'ok', guardado:true}))
        .setMimeType(ContentService.MimeType.JSON);
    } else if (data.evento) {
      const sheetActividad = obtenerOCrearHojaActividad();
      sheetActividad.appendRow([
        data.fechaHora || new Date().toLocaleString('es-AR'),
        data.evento, data.detalle || ''
      ]);
    } else {
      const sheet = obtenerOCrearHoja();
      sheet.appendRow([
        data.fecha || new Date().toLocaleDateString('es-AR'),
        data.cliente || 'N/A', data.cuit || 'N/A', data.tipo || 'N/A',
        data.neto || 0, data.iva || 0, data.retIva || 0, data.retGan || 0,
        data.percIIBB || 0, data.total || 0, data.estado || 'N/A', data.observaciones || ''
      ]);
    }
    return ContentService.createTextOutput(JSON.stringify({status:'ok', guardado:true}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch (error) {
    return ContentService.createTextOutput(JSON.stringify({status:'error', mensaje: error.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

// Instala los disparadores de digesto semanal y reporte mensual automáticamente,
// solo si todavía no existen (evita duplicados si se llama más de una vez).
function asegurarTriggers() {
  const existentes = ScriptApp.getProjectTriggers().map(t => t.getHandlerFunction());

  if (!existentes.includes('digestoSemanal')) {
    ScriptApp.newTrigger('digestoSemanal')
      .timeBased()
      .onWeekDay(ScriptApp.WeekDay.MONDAY)
      .atHour(8)
      .create();
  }

  if (!existentes.includes('reporteMensual')) {
    ScriptApp.newTrigger('reporteMensual')
      .timeBased()
      .onMonthDay(1)
      .atHour(8)
      .create();
  }
}

function obtenerOCrearHoja() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const nombreHoja = 'FiscoIA_Analisis';
  let sheet = ss.getSheetByName(nombreHoja);
  if (!sheet) {
    sheet = ss.insertSheet(nombreHoja);
    sheet.appendRow(['Fecha','Cliente','CUIT','Tipo','Neto','IVA','Ret. IVA','Ret. Ganancias','Perc. IIBB','Total','Estado','Observaciones']);
    sheet.getRange(1,1,1,12).setFontWeight('bold').setBackground('#1B3A6B').setFontColor('#FFFFFF');
    sheet.setFrozenRows(1);
  }
  return sheet;
}

function obtenerOCrearHojaActividad() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const nombreHoja = 'FiscoIA_Actividad';
  let sheet = ss.getSheetByName(nombreHoja);
  if (!sheet) {
    sheet = ss.insertSheet(nombreHoja);
    sheet.appendRow(['Fecha y Hora','Tipo de Evento','Detalle']);
    sheet.getRange(1,1,1,3).setFontWeight('bold').setBackground('#10B981').setFontColor('#1F2937');
    sheet.setFrozenRows(1);
    sheet.setColumnWidth(3, 500);
  }
  return sheet;
}

function obtenerOCrearHojaConfig() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const nombreHoja = 'FiscoIA_Config';
  let sheet = ss.getSheetByName(nombreHoja);
  if (!sheet) {
    sheet = ss.insertSheet(nombreHoja);
    sheet.getRange('A1').setValue('Email para digesto semanal:');
    sheet.getRange('A1').setFontWeight('bold');
  }
  return sheet;
}

// === DIGESTO SEMANAL ===
// Se activa solo (ver función asegurarTriggers) al guardar tu email — nada que configurar a mano.
function digestoSemanal() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheetConfig = ss.getSheetByName('FiscoIA_Config');
  const email = sheetConfig ? sheetConfig.getRange('B1').getValue() : '';
  if (!email) return; // no configuraste tu email todavía

  const sheet = ss.getSheetByName('FiscoIA_Analisis');
  if (!sheet) return;

  const filas = sheet.getDataRange().getValues();
  const hoy = new Date();
  const hace7dias = new Date(hoy.getTime() - 7*24*60*60*1000);

  let totalFacturas = 0, totalNeto = 0, totalIVA = 0, totalRet = 0;
  const clientes = new Set();

  for (let i = 1; i < filas.length; i++) {
    const fechaFila = new Date(filas[i][0]);
    if (fechaFila >= hace7dias) {
      totalFacturas++;
      clientes.add(filas[i][1]);
      totalNeto += Number(filas[i][4]) || 0;
      totalIVA += Number(filas[i][5]) || 0;
      totalRet += (Number(filas[i][6]) || 0) + (Number(filas[i][7]) || 0) + (Number(filas[i][8]) || 0);
    }
  }

  const cuerpo = `Resumen semanal de FiscoIA (últimos 7 días)

Facturas analizadas: ${totalFacturas}
Clientes distintos: ${clientes.size}
Total Neto: $${totalNeto.toLocaleString('es-AR')}
Total IVA: $${totalIVA.toLocaleString('es-AR')}
Retenciones/Percepciones: $${totalRet.toLocaleString('es-AR')}

Ver hoja completa: ${ss.getUrl()}

— FiscoIA (mensaje automático desde tu propia hoja, no pasó por servidores de LABStudio)`;

  MailApp.sendEmail(email, `📊 FiscoIA — Resumen semanal (${totalFacturas} facturas)`, cuerpo);
}

// === REPORTE MENSUAL EN PDF ===
// Se activa solo (ver función asegurarTriggers) al guardar tu email — nada que configurar a mano.
function reporteMensual() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheetConfig = ss.getSheetByName('FiscoIA_Config');
  const email = sheetConfig ? sheetConfig.getRange('B1').getValue() : '';
  if (!email) return; // no configuraste tu email todavía

  const sheet = ss.getSheetByName('FiscoIA_Analisis');
  if (!sheet) return;

  const filas = sheet.getDataRange().getValues();
  const hoy = new Date();
  // Mes anterior completo (si corre el día 1, reporta el mes que acaba de cerrar)
  const mesReporte = new Date(hoy.getFullYear(), hoy.getMonth() - 1, 1);
  const inicioMes = new Date(mesReporte.getFullYear(), mesReporte.getMonth(), 1);
  const finMes = new Date(mesReporte.getFullYear(), mesReporte.getMonth() + 1, 0, 23, 59, 59);
  const nombreMes = mesReporte.toLocaleDateString('es-AR', { month: 'long', year: 'numeric' });

  let totalFacturas = 0, totalNeto = 0, totalIVA = 0, totalRet = 0;
  const porCliente = {};

  for (let i = 1; i < filas.length; i++) {
    const fechaFila = new Date(filas[i][0]);
    if (fechaFila >= inicioMes && fechaFila <= finMes) {
      const cliente = filas[i][1] || 'Sin identificar';
      const neto = Number(filas[i][4]) || 0;
      const iva = Number(filas[i][5]) || 0;
      const ret = (Number(filas[i][6]) || 0) + (Number(filas[i][7]) || 0) + (Number(filas[i][8]) || 0);

      totalFacturas++;
      totalNeto += neto;
      totalIVA += iva;
      totalRet += ret;

      if (!porCliente[cliente]) porCliente[cliente] = { facturas: 0, neto: 0, iva: 0 };
      porCliente[cliente].facturas++;
      porCliente[cliente].neto += neto;
      porCliente[cliente].iva += iva;
    }
  }

  if (totalFacturas === 0) return; // no hubo actividad ese mes, no manda reporte vacío

  const fmt = (n) => '$' + n.toLocaleString('es-AR', { minimumFractionDigits: 2, maximumFractionDigits: 2 });

  let filasClientes = '';
  Object.keys(porCliente).sort().forEach(cliente => {
    const c = porCliente[cliente];
    filasClientes += `
      ${cliente}
      ${c.facturas}
      ${fmt(c.neto)}
      ${fmt(c.iva)}
    `;
  });

  const html = `
    
    
      

        
FiscoIA — Reporte Mensual

        
${nombreMes}

      


      

        

          
${totalFacturas}

          
Facturas analizadas

        

        

          
${Object.keys(porCliente).length}

          
Clientes distintos

        

      


      

        

          Concepto
          Monto
        

        
Total Neto${fmt(totalNeto)}

        
Total IVA${fmt(totalIVA)}

        
Retenciones/Percepciones${fmt(totalRet)}

      


      
Detalle por cliente

      
        ${filasClientes}
      

        

          Cliente
          Facturas
          Neto
          IVA
        


      

        Generado automáticamente por FiscoIA desde tu propia hoja de Google — no pasó por servidores de LABStudio.
      

      
    
    `;

  const pdfBlob = HtmlService.createHtmlOutput(html).getAs('application/pdf')
    .setName(`FiscoIA_Reporte_${mesReporte.getFullYear()}-${String(mesReporte.getMonth()+1).padStart(2,'0')}.pdf`);

  MailApp.sendEmail({
    to: email,
    subject: `📄 FiscoIA — Reporte mensual de ${nombreMes} (${totalFacturas} facturas)`,
    body: `Adjunto tu reporte mensual de ${nombreMes}.\n\nResumen rápido: ${totalFacturas} facturas, ${Object.keys(porCliente).length} clientes, ${fmt(totalNeto)} neto total.\n\nVer hoja completa: ${ss.getUrl()}\n\n— FiscoIA (mensaje automático desde tu propia hoja)`,
    attachments: [pdfBlob]
  });
}
